```python

import os

def modify_filename(path):
    for p in os.listdir(path):
        p = os.path.join(path, p)
        if os.path.isdir(p):
            modify_filename(p)
        else:
            os.rename(p, os.path.join(os.path.dirname(p), os.path.basename(p).replace('markdown', 'md')))


modify_filename("C://Users//evo//Desktop//iterate")

```
